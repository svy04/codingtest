import sys

n = int(sys.stdin.readline().strip())
result = [[0 for _ in range(100)] for _ in range(100)]
re = 0

for _ in range(n):
     x, y = map(int, sys.stdin.readline().split())
     for i in range(x, x+10):
          for j in range(y, y+10):
               result[j][i] = 1
for i in result:
     re += i.count(1)

print(re)
